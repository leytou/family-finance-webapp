> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个浏览器本地运行的家庭财务规划网页应用，支持自由添加收支项目、网格滑动选时间段、月度锚点校正、实时计算联动。

**Architecture:** Vue 3 + TypeScript + Vite 单页应用，纯前端运行。数据层使用 Vue reactive 管理状态，localStorage 持久化。计算引擎为纯函数逐月递推，输出驱动年度/月度两张表格。左右分栏布局，左侧参数面板，右侧表格视图。

**Tech Stack:** Vue 3 (Composition API), TypeScript, Vite, UnoCSS, Vitest + @vue/test-utils

```typescript
export function addMonths(yyyymm: number, offset: number): number {
  const year = Math.floor(yyyymm / 100)
  const month = yyyymm % 100
  const total = year * 12 + (month - 1) + offset
  const newYear = Math.floor(total / 12)
  const newMonth = (total % 12) + 1
  return newYear * 100 + newMonth
}

export function monthRange(start: number, end: number): number[] {
  const result: number[] = []
  let current = start
  while (compareMonth(current, end) <= 0) {
    result.push(current)
    current = addMonths(current, 1)
  }
  return result
}

export function isInRange(
  month: number,
  start: number,
  end: number
): boolean {
  return compareMonth(month, start) >= 0 && compareMonth(month, end) <= 0
}

export function compareMonth(a: number, b: number): number {
  return a - b
}

export function formatMonth(yyyymm: number): string {
  const year = Math.floor(yyyymm / 100)
  const month = yyyymm % 100
  return `${year}-${String(month).padStart(2, '0')}`
}

export function getCurrentMonth(): number {
  const now = new Date()
  return (now.getFullYear()) * 100 + (now.getMonth() + 1)
}
```

```typescript
import type { PlanData, MonthResult } from '../types'
import { addMonths, isInRange } from '../utils/month'

export function calculate(plan: PlanData): MonthResult[] {
  const { systemParams, items, anchors } = plan
  const results: MonthResult[] = []

  for (let i = 0; i < 60; i++) {
    const month = addMonths(systemParams.startMonth, i)
    const prevCum = i === 0 ? systemParams.currentSavings : results[i - 1].cumSavings

    const incomeItems: { name: string; amount: number }[] = []
    const expenseItems: { name: string; amount: number }[] = []

    for (const item of items) {
      const amount = getActiveAmount(item.segments, month)
      if (amount === 0) continue
      if (item.type === 'income') {
        incomeItems.push({ name: item.name, amount })
      } else {
        expenseItems.push({ name: item.name, amount })
      }
    }

    const totalIncome = incomeItems.reduce((s, x) => s + x.amount, 0)
    const totalExpense = expenseItems.reduce((s, x) => s + x.amount, 0)
    const investReturn = prevCum * systemParams.annualRate / 12
    const netSavings = totalIncome - totalExpense + investReturn

    const anchor = anchors.find(a => a.month === month)
    const isAnchor = !!anchor
    const cumSavings = isAnchor ? anchor.actualSavings : prevCum + netSavings

    results.push({
      month, incomeItems, expenseItems,
      totalIncome, totalExpense, investReturn,
      netSavings, cumSavings, isAnchor,
    })
  }

  return results
}

function getActiveAmount(
  segments: { amount: number; startMonth: number; endMonth: number }[],
  month: number
): number {
  let active = 0
  for (const seg of segments) {
    if (isInRange(month, seg.startMonth, seg.endMonth)) {
      active = seg.amount
    }
  }
  return active
}
```

```typescript
import { ref } from 'vue'
import type { PlanData, CashFlowItem } from '../types'
import { getCurrentMonth } from '../utils/month'

const STORAGE_KEY = 'family-finance-plan'

function createDefault(): PlanData {
  return {
    version: 1,
    systemParams: {
      currentSavings: 0,
      startMonth: getCurrentMonth(),
      annualRate: 0.025,
    },
    items: [],
    anchors: [],
  }
}

export function useStore() {
  const raw = localStorage.getItem(STORAGE_KEY)
  const data = ref<PlanData>(raw ? JSON.parse(raw) : createDefault())

  function save() {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(data.value))
  }

  function addItem(name: string, type: 'income' | 'expense') {
    const item: CashFlowItem = {
      id: Date.now().toString(36) + Math.random().toString(36).slice(2, 6),
      name,
      type,
      segments: [],
    }
    data.value.items.push(item)
    save()
  }

  function removeItem(id: string) {
    data.value.items = data.value.items.filter(i => i.id !== id)
    save()
  }

  function reset() {
    data.value = createDefault()
    localStorage.removeItem(STORAGE_KEY)
  }

  return { data, save, addItem, removeItem, reset }
}
```

```vue
<script setup lang="ts">
import { computed } from 'vue'
import type { MonthResult } from '../types'
import { formatCurrency } from '../utils/format'

const props = defineProps<{ results: MonthResult[] }>()

interface YearSummary {
  year: number
  startSavings: number
  incomeItems: { name: string; total: number }[]
  totalIncome: number
  expenseItems: { name: string; total: number }[]
  totalExpense: number
  investReturn: number
  yearBalance: number
  endSavings: number
}

const yearSummaries = computed<YearSummary[]>(() => {
  const map = new Map<number, MonthResult[]>()
  for (const r of props.results) {
    const year = Math.floor(r.month / 100)
    if (!map.has(year)) map.set(year, [])
    map.get(year)!.push(r)
  }

  const summaries: YearSummary[] = []
  for (const [year, months] of map) {
    const firstIdx = props.results.indexOf(months[0])
    const startSavings = firstIdx === 0
      ? props.results[0].cumSavings - props.results[0].netSavings
      : props.results[firstIdx - 1].cumSavings

    const incomeMap = new Map<string, number>()
    const expenseMap = new Map<string, number>()
    let investReturn = 0

    for (const m of months) {
      for (const item of m.incomeItems) {
        incomeMap.set(item.name, (incomeMap.get(item.name) ?? 0) + item.amount)
      }
      for (const item of m.expenseItems) {
        expenseMap.set(item.name, (expenseMap.get(item.name) ?? 0) + item.amount)
      }
      investReturn += m.investReturn
    }

    const incomeItems = [...incomeMap].map(([name, total]) => ({ name, total }))
    const expenseItems = [...expenseMap].map(([name, total]) => ({ name, total }))
    const totalIncome = incomeItems.reduce((s, x) => s + x.total, 0) + investReturn
    const totalExpense = expenseItems.reduce((s, x) => s + x.total, 0)
    const endSavings = months[months.length - 1].cumSavings

    summaries.push({
      year, startSavings, incomeItems, totalIncome,
      expenseItems, totalExpense, investReturn,
      yearBalance: totalIncome - totalExpense,
      endSavings,
    })
  }
  return summaries
})

const years = computed(() => yearSummaries.value.map(s => s.year))
</script>

      </thead>
      <tbody>
        <tr class="border-b">
          <td class="px-1 py-0.5">年初储蓄</td>
          <td v-for="s in yearSummaries" :key="s.year" class="px-1 py-0.5 text-right">
            {{ formatCurrency(s.startSavings) }}
          </td>

        <tr v-for="item in yearSummaries[0]?.incomeItems ?? []" :key="'i-'+item.name" class="border-b">
          <td class="px-1 py-0.5 text-green-700">{{ item.name }}</td>
          <td v-for="s in yearSummaries" :key="s.year" class="px-1 py-0.5 text-right">
            {{ formatCurrency(s.incomeItems.find(i => i.name === item.name)?.total ?? 0) }}
```

```vue
<script setup lang="ts">
import { ref, computed } from 'vue'
import { addMonths } from '../utils/month'

const props = defineProps<{
  startMonth: number
  selectedStart: number | null
  selectedEnd: number | null
}>()

const emit = defineEmits<{
  (e: 'select', range: { startMonth: number; endMonth: number }): void
}>()

const months = computed(() => {
  return Array.from({ length: 60 }, (_, i) => addMonths(props.startMonth, i))
})

const dragging = ref(false)
const dragStart = ref<number | null>(null)
const dragEnd = ref<number | null>(null)

function isSelected(month: number): boolean {
  const start = props.selectedStart
  const end = props.selectedEnd
  if (start === null || end === null) return false
  return month >= start && month <= end
}

function isDragSelected(month: number): boolean {
  if (!dragging.value || dragStart.value === null || dragEnd.value === null) return false
  const s = Math.min(dragStart.value, dragEnd.value)
  const e = Math.max(dragStart.value, dragEnd.value)
  return month >= s && month <= e
}

function onMouseDown(month: number) {
  dragging.value = true
  dragStart.value = month
  dragEnd.value = month
}

function onMouseMove(month: number) {
  if (!dragging.value) return
  dragEnd.value = month
}

function onMouseUp() {
  if (!dragging.value || dragStart.value === null || dragEnd.value === null) return
  dragging.value = false
  const s = Math.min(dragStart.value, dragEnd.value)
  const e = Math.max(dragStart.value, dragEnd.value)
  emit('select', { startMonth: s, endMonth: e })
}
</script>

<template>
  <div class="grid grid-cols-12 gap-px select-none" @mouseleave="onMouseUp">
    <div
      v-for="m in months"
      :key="m"
      data-testid="grid-cell"
      class="h-5 flex items-center justify-center text-[10px] cursor-pointer border rounded-sm transition-colors"
      :class="{
        selected: isSelected(m) || isDragSelected(m),
        'bg-blue-400 text-white': isSelected(m) || isDragSelected(m),
        'hover:bg-blue-100': !isSelected(m) && !isDragSelected(m),
      }"
      @mousedown.prevent="onMouseDown(m)"
      @mousemove="onMouseMove(m)"
      @mouseup="onMouseUp"
    >
      {{ m % 100 }}
    </div>
  </div>
</template>
```

```typescript
function addAnchor(month: number, actualSavings: number) {
  const existing = data.value.anchors.findIndex(a => a.month === month)
  if (existing >= 0) {
    data.value.anchors[existing].actualSavings = actualSavings
  } else {
    data.value.anchors.push({ month, actualSavings })
  }
  save()
}

function removeAnchor(month: number) {
  data.value.anchors = data.value.anchors.filter(a => a.month !== month)
  save()
}
```

```vue
<script setup lang="ts">
import type { MonthResult } from '../types'
import { formatCurrency } from '../utils/format'
import { formatMonth } from '../utils/month'

const props = defineProps<{
  result: MonthResult
  field: string
  x: number
  y: number
}>()

const emit = defineEmits<{ (e: 'close'): void }>()

function getFormula(): string {
  const r = props.result
  switch (props.field) {
    case 'investReturn':
      return `理财收益 = 上月累计储蓄 × 年利率 / 12`
    case 'netSavings':
      return `净储蓄 = 总收入(${formatCurrency(r.totalIncome)}) - 总支出(${formatCurrency(r.totalExpense)}) + 理财(${formatCurrency(r.investReturn)}) = ${formatCurrency(r.netSavings)}`
    case 'cumSavings':
      return r.isAnchor
        ? `锚点月份，实际储蓄 = ${formatCurrency(r.cumSavings)}`
        : `累计储蓄 = 上月累计 + 当月净储蓄(${formatCurrency(r.netSavings)})`
    default:
      return ''
  }
}
</script>

<template>
  <div
    class="fixed bg-white border rounded shadow-lg p-2 text-xs max-w-64 z-50"
    :style="{ left: x + 'px', top: y + 'px' }"
    @mouseleave="emit('close')"
  >
    <div class="font-bold mb-1">{{ formatMonth(result.month) }} - {{ field }}</div>
    <div class="text-gray-600">{{ getFormula() }}</div>
  </div>
</template>
```
